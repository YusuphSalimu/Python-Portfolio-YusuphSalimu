💻 Code Example

Simple implementation to track product prices:

from price_tracker import EcommercePriceTracker # Initialize the tracker tracker = EcommercePriceTracker(headless=True) # Search for products products = tracker.search_products("iPhone 15") # Set up price monitoring products_to_track = [ { 'name': 'iPhone 15', 'search_query': 'iPhone 15', 'target_price': 2500000 # TZS } ] # Start continuous tracking tracker.track_prices(products_to_track, duration_hours=24) # Generate price report report = tracker.generate_price_report("iPhone 15") print(f"Average price: TZS {report['average_price']:,.0f}") print(f"Price trend: {report['price_trend']}")
